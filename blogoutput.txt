        document.addEventListener('DOMContentLoaded', function() {
            // Manejar filtros de categor√≠a y orden
            const categoriaSelect = document.getElementById('categoriaSelect');
            const ordenSelect = document.getElementById('ordenSelect');
            
            if (categoriaSelect) {
                categoriaSelect.addEventListener('change', function() {
                    aplicarFiltros();
                });
            }
            
            if (ordenSelect) {
                ordenSelect.addEventListener('change', function() {
                    aplicarFiltros();
                });
            }
            
            function aplicarFiltros() {
                let url = 'blog.php';
                let params = [];
                
                const categoria = categoriaSelect ? categoriaSelect.value : '';
                const orden = ordenSelect ? ordenSelect.value : '';
                const busqueda = document.querySelector('input[name="busqueda"]').value.trim();
                
                if (categoria && categoria !== '0') {
                    params.push(`categoria=${encodeURIComponent(categoria)}`);
                }
                
                if (busqueda) {
                    params.push(`busqueda=${encodeURIComponent(busqueda)}`);
                }
                
                if (orden && orden !== 'recientes') {
                    params.push(`orden=${encodeURIComponent(orden)}`);
                }
                
                if (params.length > 0) {
                    url += '?' + params.join('&');
                }
                
                window.location.href = url;
            }
            
            // Hacer que las tarjetas de posts tengan la misma altura en cada fila
            function equalizeCardHeights() {
                // Solo ejecutar en pantallas m√°s grandes donde hay m√∫ltiples columnas
                if (window.innerWidth >= 768) {
                    const cards = document.querySelectorAll('.post-card');
                    let maxHeight = 0;
                    
                    // Restablecer alturas
                    cards.forEach(card => {
                        card.style.height = 'auto';
                        const height = card.offsetHeight;
                        maxHeight = Math.max(maxHeight, height);
                    });
                    
                    // Aplicar altura m√°xima
                    cards.forEach(card => {
                        card.style.height = maxHeight + 'px';
                    });
                } else {
                    // En m√≥vil, resetear alturas
                    const cards = document.querySelectorAll('.post-card');
                    cards.forEach(card => {
                        card.style.height = 'auto';
                    });
                }
            }
            
            // Ejecutar al cargar la p√°gina
            equalizeCardHeights();
            
            // Ejecutar al cambiar el tama√±o de la ventana
            window.addEventListener('resize', equalizeCardHeights);
        });
